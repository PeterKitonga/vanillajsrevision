In JavaScript, Document Object Model (DOM) is a programming interface for HTML and XML documents. The DOM represents the document as nodes and objects. More info here >.

Code:

  1. var app = {};
  2.  
  3. app.number = 1;
  4.  
  5. /*
  6. * Using the document object available in JavaScript we can
  7. * select specific elements in the DOM perform various actions to it
  8. */
  9. app.element = document.getElementById('link');
  10.  
  11. /*
  12. * There are various properties and methods available to an 
  13. * element that had been selected. This properties and method help
  14. * manipulate the element in the DOM
  15. */
  16. console.log('Link href attribute: ' + app.element.getAttribute('href'));
  17.  
  18. console.log('X offset of element in pixels: ' + app.element.offsetLeft);
  19.  
  20. console.log('Y offset of element in pixels: ' + app.element.offsetTop);
  21.  
  22. app.interval = setInterval(function() {
  23.     if (app.number === 1) {
  24.         console.log('Text: "' + app.element.innerText + '" will change in 10 seconds');
  25.     }
  26.  
  27.     if (app.number === 10) {
  28.         app.element.innerText = 'Element Changed';
  29.         console.log('Text for link is: "' + app.element.innerText + '"');
  30.         console.log('Text: "' + app.element.innerText + '" will bolden in 10 seconds');
  31.     }
  32.  
  33.     if (app.number === 20) {
  34.         app.element.innerHTML = '<strong>Element Bold</strong>';
  35.         console.log('Content for link is: "' + app.element.innerHTML + '"');
  36.         clearInterval(app.interval);
  37.     }
  38.  
  39.     app.number++;
  40. }, 1000);
