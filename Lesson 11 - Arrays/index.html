In JavaScript, an array is a data type that is used to store a list of values or objects. They can be used to describe list of items from a database table or values of a series of calculations.

Code:

  1. var names = ['Peter'];
  2.  
  3. // adding items to the array
  4. names[1] = 'Dex';
  5. console.log('Adding item via index: ' + JSON.stringify(names));
  6.  
  7. // this method can add multiple items to the array
  8. names.push('Wayne', 'Fred', 'Jax');
  9. console.log('Adding items via push: ' + JSON.stringify(names));
  10.  
  11. // updating an item in the array
  12. names[1] = 'Max';
  13. console.log('Updating item via index: ' + JSON.stringify(names));
  14.  
  15. // changes Fred to Darrel
  16. names.splice(3, 1, 'Darrel');
  17. console.log('Updating item via splice: ' + JSON.stringify(names));
  18.  
  19. // removing items from array
  20. names.splice(names.indexOf('Peter'), 1);
  21. console.log('Removing item: ' + JSON.stringify(names));
  22.  
  23. // adding items with named indexes
  24. names['color'] = 'black';
  25. console.log('Adding items with named indexes: ');
  26. console.log(names);
