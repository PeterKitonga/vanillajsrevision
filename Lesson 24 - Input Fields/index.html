In JavaScript, we can use event handlers to get the value of input fields in the page and do dynamic things like search or validation. In this lesson I learnt how to select an input field and fetch it's value inorder to print the output in a span element.

Code:

  1. var app = {};
  2.  
  3. // Reduces redundant declaration of getElementById
  4. app.getElement = function (element) {
  5.     return document.getElementById(element);
  6. }
  7.  
  8. // Get elements required
  9. app.entry = app.getElement('result-input');
  10. app.output = app.getElement('result-output');
  11.  
  12. // Create event handler
  13. app.entry.addEventListener('keyup', function (event) {
  14.     // event.target.value gets the value of the text input
  15.     if (event.target.value.length > 0) {
  16.         app.output.innerText = event.target.value;
  17.     } else {
  18.         app.output.innerText = 'Start typing again...';
  19.     }
  20. });
