In JavaScript, intervals are created using the setInterval function which repeatedly calls a function or executes a snippet of code within a fixed time delay. This interval is usually cleared by using the clearInterval function whose parameter will be the variable assigned to the interval. More info here >.

Code:

  1. var app = {};
  2.  
  3. app.countdown = function(settings) {
  4.     var interval,
  5.         counter = 0,
  6.         // Start and end points
  7.         start_at = 0,
  8.         end_at = 0,
  9.         message = '';
  10.  
  11.     // Validate the function parameters
  12.     if(settings === undefined) {
  13.         message = 'Please provide settings';
  14.     } else {
  15.         // Check if one parameter is missing
  16.         if (settings.start_at === undefined || settings.end_at === undefined) {
  17.             /*
  18.             * Below is a ternary operator which is similar to an if...else statement.
  19.             * More info: https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Operators/Conditional_Operator
  20.             */
  21.             message = settings.hasOwnProperty('start_at') ? 'Please provide end_at setting' : 'Please provide start_at setting';
  22.         } else {
  23.             start_at = parseInt(settings.start_at);
  24.             end_at = parseInt(settings.end_at);
  25.  
  26.             // Check if the parameters are numbers
  27.             if (!isNaN(start_at) && !isNaN(end_at)) {
  28.                 message = 'Begin countdown from ' + start_at + ' to ' + end_at;
  29.                 counter = start_at;
  30.  
  31.                 interval = setInterval(function() {
  32.                     // Check if the counter has reached the end
  33.                     if (counter < end_at) {
  34.                         console.log('Finished countdown');
  35.                         clearInterval(interval);
  36.                     } else {
  37.                         console.log('Countdown at: ' + counter);
  38.                     }
  39.  
  40.                     counter--;
  41.                 }, 1000);
  42.             } else {
  43.                 message = 'Please provide start_at and end_at settings as numbers';
  44.             }
  45.         }
  46.     }
  47.  
  48.     return message;
  49. };
