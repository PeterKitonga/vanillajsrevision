Query Selector

Next >>

In JavaScript, the querySelector() and querySelectorAll() methods are used to select elements from the DOM using CSS style selectors like nav, div, ul e.t.c. These methods can also be used to select elements with the specified class or id.

Code:

  1. var app = {};
  2.  
  3. /*
  4. * Fetches the first item with the selector defined.
  5. * In the example below it will select the first 'a' tag
  6. * inside the nav element.
  7. * More info: https://developer.mozilla.org/en-US/docs/Web/API/Document/querySelector
  8. */
  9. app.element = document.querySelector('nav ul li a');
  10.  
  11. console.log('Query selector result: ' + app.element);
  12.  
  13. /*
  14. * Here, we can select an element using it's id.
  15. */
  16. app.element_id = document.querySelector('a#home');
  17.  
  18. console.log('Query selector by id result: ' + app.element_id);
  19.  
  20. /*
  21. * Here, we can select an element using it's class.
  22. */
  23. app.element_class = document.querySelector('a.contact-link');
  24.  
  25. console.log('Query selector by class result: ' + app.element_class);
  26.  
  27. /*
  28. * This method selects all items with the selector defined. 
  29. * More info: https://developer.mozilla.org/en-US/docs/Web/API/Document/querySelectorAll
  30. */
  31. app.elements = document.querySelectorAll('nav ul li a');
  32.  
  33. if (app.elements.length > 0) {
  34.     for (index = 0; index < app.elements.length; index++) {
  35.         var item = app.elements[index];
  36.         console.log('Query selector all result: ' + item.text + ' - ' + item.getAttribute('href'));
  37.     }
  38. }
