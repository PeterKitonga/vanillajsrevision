In JavaScript, a callback is a function passed into another function as an argument. This function is then called to perform a specific action. More info here >.

Code:

  1. var app = {};
  2.  
  3. app.add = function (numbers, _callback) {
  4.     var result = 0;
  5.  
  6.     // validate if the numbers are defined
  7.     if (numbers !== undefined && numbers.length > 0) {
  8.         // loop through and sum the numbers
  9.         for(number in numbers) {
  10.             result += numbers[number];
  11.         }
  12.  
  13.         // checks if the callback is defined
  14.         if (_callback !== undefined) {
  15.             // if defined pass a parameter and run the defined callback
  16.             return _callback(result);
  17.         }
  18.  
  19.         return 'Result is: ' + result;
  20.     } else {
  21.         return 'Please provide numbers';
  22.     }
  23. }
  24.  
  25. /*
  26. * Running the code
  27. */
  28. <script>
  29.     var example_one = app.add([5, 6, 10, 21]);
  30.  
  31.     console.log(example_one);
  32.  
  33.     var example_two = app.add([5, 6, 10, 21], function (sum) {
  34.         return 'Custom callback result: ' + (sum + 20);
  35.     });
  36.  
  37.     console.log(example_two);
  38. </script>
