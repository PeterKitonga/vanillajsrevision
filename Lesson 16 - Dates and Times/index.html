A JavaScript date is fundamentally specified as the number of milliseconds that have elapsed since midnight on January 1, 1970, UTC. More info here >.

Code:

  1. /*
  2. * A javascript date object is instatiated by the Date() construct and can be used to 
  3. * fetch the date of a single moment in time in a platform-independent format.
  4. */
  5. var app = {
  6.     date: new Date(),
  7.     today: function() {
  8.         return this.date.toLocaleDateString('en-GB', {timeZone: 'Africa/Nairobi', timeZoneName: 'long'});
  9.     },
  10.     now_formatted: function() {
  11.         return this.date.toLocaleString('en-GB', {timeZone: 'Africa/Nairobi'});
  12.     },
  13.     format_date: function(date) {
  14.         return new Date(date).toLocaleString('en-GB', {timeZone: 'Africa/Nairobi'});
  15.     }
  16. };
  17.  
  18. /*
  19. * Using the .toLocalDateString() prototype we can fetch today's date and format it to the specific timezone and format
  20. */
  21. console.log('Date today is: ' + app.today());
  22.  
  23. /*
  24. * Here we can pass a UTC timestamp to be formatted.
  25. */
  26. console.log('Formatted date is: ' + app.format_date(Date.UTC(2012, 11, 20, 3, 0, 0)));
  27.  
  28. /*
  29. * Here we can pass the date as a string to be formatted
  30. */
  31. console.log('Formatted date is: ' + app.format_date('2020-08-28 19:18:00'));
